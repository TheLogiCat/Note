\node[draw=BrandBlue, rounded corners=5pt, fill=white, very thick, align=left, inner sep=4mm, text=Gray, minimum width=150mm] (eqs) at ($(bg.center)+(0,40mm)$) {%
\begin{minipage}{145mm}
\textbf{沿射线} $\mathbf r(t)=\mathbf o + t\mathbf d,\ t\in[t_n,t_f]$\\[1mm]
\textbf{像素颜色：}
\[
C(\mathbf r)=\int_{t_n}^{t_f} T(t)\,\sigma(\mathbf r(t))\,\mathbf c(\mathbf r(t),\mathbf d)\,dt
\]
\textbf{透过率：}
\[
T(t)=\exp\!\left(-\int_{t_n}^{t}\sigma(\mathbf r(s))\,ds\right)
\]
\textbf{离散 $\alpha$ 合成：}
\[
\alpha_i=1-e^{-\sigma_i\Delta t_i},\quad
w_i=\alpha_i\!\!\prod_{j<i}(1-\alpha_j),\quad
C\approx\sum_i w_i\,\mathbf c_i
\]
\end{minipage}
};
